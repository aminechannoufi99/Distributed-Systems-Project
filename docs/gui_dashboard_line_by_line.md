# gui/dashboard.py line by line

Note: blank lines are just for readability.

- `from __future__ import annotations` -> active les annotations de types en avant.
- `import queue` -> pour queue.Empty.
- `import time` -> timestamps pour logs.
- `import tkinter as tk` -> base Tkinter.
- `from tkinter import ttk` -> widgets ttk.
- `from typing import Callable, Dict, Iterable` -> types utilises.
- `import messages` -> protocole messages.
- `from gui.panels import OrdersPanel, PizzaioliPanel` -> panneaux GUI.
- `class Dashboard:` -> classe principale GUI.
- `    def __init__(` -> constructeur.
- `        self,` -> instance.
- `        event_queue: "queue.Queue",` -> file d evenements GUI.
- `        pizzaiolo_ids: Iterable[int],` -> ids a afficher.
- `        on_new_order: Callable[[], None],` -> callback bouton.
- `        on_close: Callable[[], None],` -> callback fermeture.
- `    ) -> None:` -> fin signature.
- `        self.event_queue = event_queue` -> stocke la file.
- `        self.on_new_order = on_new_order` -> stocke callback.
- `        self.on_close = on_close` -> stocke callback.
- `        self.root = tk.Tk()` -> cree la fenetre.
- `        self.root.title("Pizza Delivery - Monitoring")` -> titre.
- `        self.root.geometry("940x640")` -> taille.
- `        self.root.protocol("WM_DELETE_WINDOW", self._handle_close)` -> gestion fermeture.
- `        self._setup_style()` -> applique style.
- `        self._build_layout(list(pizzaiolo_ids))` -> construit layout.
- `    def run(self) -> None:` -> lance la GUI.
- `        self._poll_events()` -> demarre lecture des events.
- `        self.root.mainloop()` -> boucle Tkinter.
- `    def _setup_style(self) -> None:` -> configure theme.
- `        style = ttk.Style(self.root)` -> objet style.
- `        if "clam" in style.theme_names():` -> si clam existe.
- `            style.theme_use("clam")` -> utilise clam.
- `    def _build_layout(self, pizzaiolo_ids: Iterable[int]) -> None:` -> layout.
- `        header = ttk.Frame(self.root)` -> frame header.
- `        header.pack(side="top", fill="x")` -> placement.
- `        title = ttk.Label(header, text="Pizza Delivery System", font=("Segoe UI", 16, "bold"))` -> titre.
- `        title.pack(side="left", padx=12, pady=10)` -> placement titre.
- `        controls = ttk.Frame(header)` -> frame controles.
- `        controls.pack(side="right", padx=12, pady=10)` -> placement controles.
- `        new_order_btn = ttk.Button(controls, text="New Order", command=self.on_new_order)` -> bouton.
- `        new_order_btn.grid(row=0, column=0, padx=6)` -> placement bouton.
- `        body = ttk.Frame(self.root)` -> frame principal.
- `        body.pack(side="top", fill="both", expand=True)` -> placement.
- `        left = ttk.Frame(body)` -> colonne gauche.
- `        left.pack(side="left", fill="y")` -> placement.
- `        right = ttk.Frame(body)` -> colonne droite.
- `        right.pack(side="right", fill="both", expand=True)` -> placement.
- `        self.pizzaioli_panel = PizzaioliPanel(left, pizzaiolo_ids)` -> panneau pizzaioli.
- `        self.pizzaioli_panel.pack(side="top", fill="x")` -> placement.
- `        self.orders_panel = OrdersPanel(right)` -> panneau commandes.
- `        self.orders_panel.pack(side="top", fill="both", expand=True)` -> placement.
- `        log_frame = ttk.Frame(right)` -> frame logs.
- `        log_frame.pack(side="bottom", fill="both", expand=False, padx=8, pady=6)` -> placement.
- `        log_label = ttk.Label(log_frame, text="Coordinator Log", font=("Segoe UI", 11, "bold"))` -> label log.
- `        log_label.pack(side="top", anchor="w")` -> placement.
- `        self.log_text = tk.Text(log_frame, height=8, wrap="word", state="disabled")` -> zone texte.
- `        self.log_text.pack(side="left", fill="both", expand=True)` -> placement.
- `        log_scroll = ttk.Scrollbar(log_frame, orient="vertical", command=self.log_text.yview)` -> scrollbar.
- `        log_scroll.pack(side="right", fill="y")` -> placement.
- `        self.log_text.configure(yscrollcommand=log_scroll.set)` -> relie scroll.
- `    def _handle_close(self) -> None:` -> gestion fermeture.
- `        self.on_close()` -> appelle callback.
- `        self.root.destroy()` -> ferme fenetre.
- `    def _poll_events(self) -> None:` -> boucle lecture events.
- `        try:` -> lecture non bloquante.
- `            while True:` -> lit tout ce qui est dispo.
- `                event = self.event_queue.get_nowait()` -> lit un event.
- `                self._handle_event(event)` -> traite event.
- `        except queue.Empty:` -> si plus rien.
- `            pass` -> ignore.
- `        self.root.after(200, self._poll_events)` -> relance apres 200ms.
- `    def _handle_event(self, event: Dict[str, object]) -> None:` -> route event.
- `        event_type = event.get("type")` -> type d event.
- `        if event_type == messages.GUI_STATE:` -> snapshot.
- `            snapshot = event.get("snapshot", {})` -> lit snapshot.
- `            self._update_state(snapshot)` -> maj GUI.
- `        elif event_type == messages.GUI_LOG:` -> log.
- `            self._append_log(event.get("message", ""))` -> ajoute log.
- `    def _update_state(self, snapshot: Dict[str, object]) -> None:` -> maj etat affichage.
- `        pizzaioli = snapshot.get("pizzaioli", {})` -> statuts pizzaioli.
- `        orders = snapshot.get("orders", {})` -> commandes.
- `        total_orders = len(orders)` -> nb commandes (info).
- `        self.pizzaioli_panel.update_statuses(pizzaioli)` -> maj pizzaioli.
- `        self.orders_panel.update_orders(orders)` -> maj commandes.
- `    def _append_log(self, message: str) -> None:` -> ajoute un log.
- `        timestamp = time.strftime("%H:%M:%S")` -> heure lisible.
- `        line = f"[{timestamp}] {message}\\n"` -> formate la ligne.
- `        self.log_text.configure(state="normal")` -> active edition.
- `        self.log_text.insert("end", line)` -> ajoute texte.
- `        self.log_text.configure(state="disabled")` -> bloque edition.
- `        self.log_text.see("end")` -> scroll en bas.
